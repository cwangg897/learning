그림판을보니까 범위가아니면 무조건 2번으로 갈 수 있었다.
그래서 이와같이 풀었다. 중요한건 비밀번호랑 현재위치랑 같은경우 계산을 해주면안된다는것이다

```java
import java.awt.*;
import java.util.*;

class Solution {
    int[] dx = {-1, 1, 0, 0, -1, -1, 1, 1}; // 상, 하, 좌, 우, 좌상, 우상, 좌하, 우하
    int[] dy = {0, 0, -1, 1, -1, 1, -1, 1}; // 상, 하, 좌, 우, 좌상, 우상, 좌하, 우하

    public int solution(int[] keypad, String password) {
        int answer = 0;
        int[][] map = new int[3][3];
        Map<Integer, Point> hashMap = new HashMap<>();
        int idx = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                map[i][j] = keypad[idx];
                hashMap.put(keypad[idx], new Point(i,j));
                idx++;
            }
        }
        // 비밀번호의 첫 숫자에서 시작
        int count = 0;
        int start = Character.getNumericValue(password.charAt(0)); // 7번
        for(int i=1; i<password.length(); i++){
            int number = Character.getNumericValue(password.charAt(i));
            Point point = hashMap.get(start);
            if(number == start){
                continue;
            }
            boolean flag = false;
            for(int j=0; j<dx.length; j++){
                int nx = point.x + dx[j];
                int ny = point.y + dy[j];
                if(nx < 0 || nx>=3 || ny<0 ||  ny>=3){
                    continue;
                }
                if(map[nx][ny] == number){
                    flag = true;
                    break;
                }
            }
            if(flag){
                count+=1;
            }else{
                count+=2;
            }
            start = number;
        }

        answer = count;

        return answer;
    }

    static class Point{
        int x;
        int y;
        public Point(int x, int y){
            this.x = x;
            this.y = y;
        }
    }

    public static void main(String[] args) {
        Solution T = new Solution();
        System.out.println(T.solution(new int[] {2, 5, 3, 7, 1, 6, 4, 9, 8}, "7596218"));
        System.out.println(T.solution(new int[]{1, 5, 7, 3, 2, 8, 9, 4, 6}, "63855526592"));
        System.out.println(T.solution(new int[]{2, 9, 3, 7, 8, 6, 4, 5, 1}, "323254677"));
        System.out.println(T.solution(new int[]{1, 6, 7, 3, 8, 9, 4, 5, 2}, "3337772122"));
    }
}
```
